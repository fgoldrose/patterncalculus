0:0 1:1

digit: (0 | 1)

succ: ((0 -> 1)| (1->1 0))
number: (rec (number->(digit| (rec number) digit)))

    (succ n
    |   (num lastdigit):n
        d: (succ lastdigit)
        ((digit-> (num d))
        | _ -> rec add1 num 0
        ) d
    )))
